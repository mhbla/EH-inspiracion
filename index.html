async function loadEntries() { try { const response = await fetch("https://opensheet.elk.sh/YOUR_SHEET_ID/Sheet1"); const data = await response.json(); const entriesContainer = document.getElementById("entries"); entriesContainer.innerHTML += data.reverse().map(entry => `
${entry.platform} — ${entry.title}
${entry.description}
Rating: ${['❄️','☃️','⛅','☀️','🔥'][parseInt(entry.rating)-1] || ''}
`).join(''); } catch (err) { console.error("Failed to load entries", err); } } loadEntries();
